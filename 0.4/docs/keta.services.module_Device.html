<inner> $reset() → {undefined}

Resets a DeviceInstance to it's $pristine state.

Source:
  • services/device.js, line 238
Returns:
undefined nothing
Example
angular.module('exampleApp', ['keta.services.Device'])
    .controller('ExampleController', function(Device) {
        var device = Device.create({
            guid: 'guid',
            tagValues: {
                IdName: {
                    name: 'IdName',
                    value: 'Device',
                    oca: 0,
                    timestamp: 123456789
                }
            }
        });
        device.tagValues.IdName.value = 'Modified Device';
        device.$update()
            .then(function(reply) {
                // success handler
                // ...
            }, function(reply) {
                // error handler
                device.$reset();
            });
    });
