Example
angular.module('exampleApp', ['keta.services.Application'])
    .controller('ExampleController', function(Application) {
        var application = Application.create(eventBus, {
            appId: 'company.app',
            entryUri: 'https://...',
            redirectUri: 'https://...',
            iconUri: 'https://...',
            names: {
                en: 'Company App',
                de: 'Firmenanwendung'
            }
        });
    });
